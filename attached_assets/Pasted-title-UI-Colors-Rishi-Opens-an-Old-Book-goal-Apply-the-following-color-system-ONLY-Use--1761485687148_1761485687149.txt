title: "UI Colors — Rishi Opens an Old Book"
goal: >
  Apply the following color system ONLY. Use these hex values as-is.
  Prioritize warmth and legible contrast over the forest backdrop.

tokens:
  # Backgrounds / Surfaces
  bg_app:         "#1C120C"     # deep warm umber (page backdrop behind everything)
  bg_header:      "#2A1B12"     # top bar / nav
  surface_panel:  "#EEE2C3"     # parchment panel (cards, modals, search)
  surface_alt:    "#D9C9A1"     # secondary parchment (dropdowns, tooltips)
  surface_ghost:  "#F7EFDA"     # ultra-light paper (inputs, chips)

  # Text / Icons
  ink_primary:    "#1F3D5B"     # main ink
  ink_secondary:  "#3D2B1F"     # warm copy on light parchment
  ink_muted:      "#6F5D4A"     # helper text
  ink_inverse:    "#F5E6C8"     # text on dark bars

  # Brand / Accents
  accent_warm:    "#C85A1E"     # primary accent (buttons, active states)
  accent_brass:   "#A38D5D"     # secondary accent (borders, tabs)
  accent_moss:    "#6B7D5B"     # tertiary accent (success/ok)
  accent_ember:   "#7A3B1E"     # pressed/active warm

  # Strokes / Dividers / Shadows
  stroke_soft:    "#BFAF86"     # hairlines on parchment
  stroke_hard:    "#5C4631"     # separators on dark areas
  shadow_tint:    "rgba(62,44,32,0.28)"  # drop shadows (warm)

  # Choropleth (ink-on-parchment, low-sat blue scale)
  choro_L0:       "#EBD8A7"     # very light (0–10%)
  choro_L1:       "#B9C7CC"     # light (10–30%)
  choro_L2:       "#8EA6B3"     # mid (30–60%)
  choro_L3:       "#5E7E93"     # dark (60–85%)
  choro_L4:       "#2E4057"     # deepest (85–100%)

  # UI States
  hover_warm:     "#D3662B"
  active_warm:    "#A94918"
  focus_ring:     "#A38D5D"
  success:        "#6B7D5B"
  warning:        "#D29B2B"
  danger:         "#8C2F26"

usage:
  header:
    bg: bg_header
    text: ink_inverse
    icon: ink_inverse
    border_bottom: stroke_hard

  buttons:
    primary_bg:  accent_warm
    primary_text: "#FFF"
    primary_hover: hover_warm
    primary_active: active_warm
    secondary_bg: surface_panel
    secondary_text: ink_primary
    secondary_border: accent_brass

  inputs:
    bg: surface_ghost
    text: ink_primary
    placeholder: ink_muted
    border: stroke_soft
    focus_border: focus_ring

  cards:
    bg: surface_panel
    text: ink_secondary
    border: stroke_soft
    shadow: shadow_tint

  map_choropleth:
    gradient: [choro_L0, choro_L1, choro_L2, choro_L3, choro_L4]
    stroke: "rgba(43,29,20,0.28)"   # sepia linework
    fill_opacity: 0.82
    blend_mode: "multiply"

  ui_accents:
    link: accent_warm
    link_hover: hover_warm
    tag_bg: surface_alt
    tag_text: ink_primary
    badge_warn_bg: warning
    badge_danger_bg: danger
    badge_text: "#FFF"

notes: >
  Keep contrast ≥ 4.5:1 for body text. Avoid pure white over parchment; use #FFF only
  on warm accents or dark bars. Do not introduce new hues.
